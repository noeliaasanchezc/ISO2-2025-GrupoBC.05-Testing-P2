6       public String calcularTarifaAerea(
7               int edad,
8               int vuelosAnuales,
9               boolean esEstudianteFuera,
10              boolean trabaja,
11              boolean viveConPadres,
12              String clase,
13              String destino,
14              double ingresos,
15              boolean viajaConNiños) {
16  
17          
18          // 1) Tarifa Pajarillo
19          
20          if (edad < 18 && vuelosAnuales >= 6) {
21              return "Pajarillo";
22          }
23  
24          
25          // 2) Tarifa Gorrión
26          
27          if (edad >= 18 && edad <= 25 &&
28                  esEstudianteFuera &&
29                  clase.equalsIgnoreCase("turista") &&
30                  vuelosAnuales >= 12) {
31              return "Gorrión";
32          }
33  
34          
35          // 3) Tarifa Viaja ahora que puedes / Atreviéndose a saltar del Nido
36         
37          if (edad >= 18 && edad <= 25 && trabaja) {
38  
39              if (viveConPadres &&
40                      clase.equalsIgnoreCase("turista") &&
41                      vuelosAnuales >= 3) {
42                  return "Viaja ahora que puedes";
43              }
44  
45              if (!viveConPadres) {
46                  return "Atreviéndose a saltar del Nido";
47              }
48          }
49  
50        
51          // 4) Tarifa Conoce Europa / Conoce Europa con tus peques
52         
53          if (edad > 25 &&
54                  ingresos > 20000 && ingresos < 35000 &&
55                  vuelosAnuales >= 6 &&
56                  clase.equalsIgnoreCase("turista") &&
57                  destino.equalsIgnoreCase("Europa")) {
58  
59              if (viajaConNiños) {
60                  return "Conoce Europa con tus peques";
61              } else {
62                  return "Conoce Europa";
63              }
64          }
65  
66          
67          // 5) Conoce el Mundo / Conoce el Mundo con tus peques
68         
69          if (edad > 25 &&
70                  ingresos > 35000 &&
71                  vuelosAnuales >= 6 &&
72                  clase.equalsIgnoreCase("business") &&
73                  (destino.equalsIgnoreCase("Asia") ||
74                          destino.equalsIgnoreCase("America"))) {
75  
76              if (viajaConNiños) {
77                  return "Conoce el Mundo con tus peques";
78              } else {
79                  return "Conoce el Mundo";
80              }
81          }
82  
83          
84          // Si no cumple ninguna condición
85          
86          return "Sin tarifa disponible";
87      }
88  }
